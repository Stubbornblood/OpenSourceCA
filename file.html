#!/bin/bash # Get motherboard information using lspci echo "

Motherboard Information

" sudo lspci | grep -i 'Host bridge\|PCI bridge\|ISA bridge\|SMBus\|USB controller\|Serial controller\|SATA controller\|IDE interface\|Audio device\|VGA compatible controller\|Ethernet controller' | xargs printf "

%s

\n" # Get network information using ifconfig echo "

Network Information

" sudo ifconfig | xargs printf "

%s

\n" # Get storage device information using lsblk echo "

Storage Device Information

" sudo lsblk -o NAME,SIZE,TYPE,MOUNTPOINT | xargs printf "

%s

\n" # Get display information using xrandr echo "

Display Information

" sudo xrandr | xargs printf "

%s

\n"
